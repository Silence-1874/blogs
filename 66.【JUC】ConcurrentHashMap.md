## Node 内部类
```java
// 相比普通的 HashMap ，对 val 与 next 加上了 volatile 修饰，添加了 find 方法
static class Node<K,V> implements Map.Entry<K,V> {
    final int hash;
    final K key;
    // 使用 volatile 修饰，以配合 CAS 使用
    volatile V val;
    volatile Node<K,V> next;

    Node(int hash, K key, V val, Node<K,V> next) {
        this.hash = hash;
        this.key = key;
        this.val = val;
        this.next = next;
    }

    public final K getKey()       { return key; }
    public final V getValue()     { return val; }
    public final int hashCode()   { return key.hashCode() ^ val.hashCode(); }
    public final String toString(){ return key + "=" + val; }
    // 默认不支持此方法
    public final V setValue(V value) {
        throw new UnsupportedOperationException();
    }

    public final boolean equals(Object o) {
        Object k, v, u; Map.Entry<?,?> e;
        return ((o instanceof Map.Entry) &&
                (k = (e = (Map.Entry<?,?>)o).getKey()) != null &&
                (v = e.getValue()) != null &&
                (k == key || k.equals(key)) &&
                (v == (u = val) || v.equals(u)));
    }

    // 寻找键等于 k 的 Node
    Node<K,V> find(int h, Object k) {
        Node<K,V> e = this;
        if (k != null) {
            do {
                K ek;
                if (e.hash == h &&
                    ((ek = e.key) == k || (ek != null && k.equals(ek))))
                    return e;
            } while ((e = e.next) != null);
        }
        return null;
    }
}
```

## ForwardingNode 内部类
```java
// 扩容时标记扩容的链表
static final class ForwardingNode<K,V> extends Node<K,V> {
    final Node<K,V>[] nextTable;
    ForwardingNode(Node<K,V>[] tab) {
        super(MOVED, null, null, null);
        this.nextTable = tab;
    }

    Node<K,V> find(int h, Object k) {
        outer: for (Node<K,V>[] tab = nextTable;;) {
            Node<K,V> e; int n;
            if (k == null || tab == null || (n = tab.length) == 0 ||
                (e = tabAt(tab, (n - 1) & h)) == null)
                return null;
            for (;;) {
                int eh; K ek;
                if ((eh = e.hash) == h &&
                    ((ek = e.key) == k || (ek != null && k.equals(ek))))
                    return e;
                if (eh < 0) {
                    if (e instanceof ForwardingNode) {
                        tab = ((ForwardingNode<K,V>)e).nextTable;
                        continue outer;
                    }
                    else
                        return e.find(h, k);
                }
                if ((e = e.next) == null)
                    return null;
            }
        }
    }
}
```

```java
public ConcurrentHashMap() {

public ConcurrentHashMap(int initialCapacity) {
    if (initialCapacity < 0)
        throw new IllegalArgumentException();
    int cap = ((initialCapacity >= (MAXIMUM_CAPACITY >>> 1)) ?
                MAXIMUM_CAPACITY :
                tableSizeFor(initialCapacity + (initialCapacity >>> 1) + 1));
    this.sizeCtl = cap;
}

public ConcurrentHashMap(Map<? extends K, ? extends V> m) {
    this.sizeCtl = DEFAULT_CAPACITY;
    putAll(m);
}

public ConcurrentHashMap(int initialCapacity, float loadFactor) {
    this(initialCapacity, loadFactor, 1);
}

// initialCapacity:初始容量
// loadFactor:负载因子
// concurrencyLevel:并发度
public ConcurrentHashMap(int initialCapacity,
                            float loadFactor, int concurrencyLevel) {
    if (!(loadFactor > 0.0f) || initialCapacity < 0 || concurrencyLevel <= 0)
        throw new IllegalArgumentException();
    // 初始容量必须不小于并发度
    if (initialCapacity < concurrencyLevel)
        initialCapacity = concurrencyLevel;
    // 计算哈希表初始大小
    long size = (long)(1.0 + (long)initialCapacity / loadFactor);
    // 将哈希表初始大小设置为第一个大于它的2的整数次幂
    // 真正的哈希表是懒加载的
    int cap = (size >= (long)MAXIMUM_CAPACITY) ?
        MAXIMUM_CAPACITY : tableSizeFor((int)size);
    this.sizeCtl = cap;
}
```

### 初始化
#### initTable()
```java
private final Node<K,V>[] initTable() {
    Node<K,V>[] tab; int sc;
    while ((tab = table) == null || tab.length == 0) {
        // 第一个线程已经把 CAS 改为 -1，第一个线程初始化未完成，其他的线程再进来都会 yield
        if ((sc = sizeCtl) < 0)
            Thread.yield();
        // 第一个线程进入后，使用 CAS 操作将 sizeCtl 改为 -1
        // 如果已经初始化成功，其他线程也能够成功完成 CAS，但在最后的 finally 中会将 sizeCtl 又恢复到之前的值
        else if (U.compareAndSwapInt(this, SIZECTL, sc, -1)) {
            try {
                if ((tab = table) == null || tab.length == 0) {
                    // 如果 sizeCtl 为 0，则使用初始容量 16
                    int n = (sc > 0) ? sc : DEFAULT_CAPACITY;
                    @SuppressWarnings("unchecked")
                    // 创建哈希表
                    Node<K,V>[] nt = (Node<K,V>[])new Node<?,?>[n];
                    table = tab = nt;
                    // 计算扩容阈值（容量的3/4)
                    sc = n - (n >>> 2);
                }
            } finally {
                sizeCtl = sc;
            }
            break;
        }
    }
    return tab;
}
```

#### putVal(K key, V value, boolean onlyIfAbsent)
```java
final V putVal(K key, V value, boolean onlyIfAbsent) {
    // 不允许
    if (key == null || value == null) throw new NullPointerException();
    // 二次哈希
    int hash = spread(key.hashCode());
    int binCount = 0;
    for (Node<K,V>[] tab = table;;) {
        // f:链表头结点
        // fh:链表头结点的哈希值
        // i:链表在哈希表中的索引
        Node<K,V> f; int n, i, fh;
        // 如果尚未初始化，调用 initTable 方法初始化哈希表
        if (tab == null || (n = tab.length) == 0)
            tab = initTable();
        // 如果链表还没初始化，创建链表
        else if ((f = tabAt(tab, i = (n - 1) & hash)) == null) {
            // 使用 CAS 操作创建链表头结点
            if (casTabAt(tab, i, null,
                            new Node<K,V>(hash, key, value, null)))
                break;
        }
        // 头结点哈希为 MOVED(-1)，表示头结点是 ForwardingNode 结点
        else if ((fh = f.hash) == MOVED)
            // 帮忙扩容
            tab = helpTransfer(tab, f);
        else {
            V oldVal = null;
            // 锁住头结点
            synchronized (f) {
                if (tabAt(tab, i) == f) {
                    // 哈希值不小于零，说明是链表的结点
                    if (fh >= 0) {
                        binCount = 1;
                        // 遍历链表
                        for (Node<K,V> e = f;; ++binCount) {
                            K ek;
                            // 找到键相同的结点，覆盖
                            if (e.hash == hash &&
                                ((ek = e.key) == key ||
                                    (ek != null && key.equals(ek)))) {
                                oldVal = e.val;
                                if (!onlyIfAbsent)
                                    e.val = value;
                                break;
                            }
                            Node<K,V> pred = e;
                            // 链表中找不到键相同的结点，在尾部插入新结点
                            if ((e = e.next) == null) {
                                pred.next = new Node<K,V>(hash, key,
                                                            value, null);
                                break;
                            }
                        }
                    }
                    // 如果结点是红黑树结点
                    else if (f instanceof TreeBin) {
                        Node<K,V> p;
                        binCount = 2;
                        // 将结点插入红黑树
                        // putTreeVal:寻找结点，若不存在则添加，并返回 null
                        if ((p = ((TreeBin<K,V>)f).putTreeVal(hash, key,
                                                        value)) != null) {
                            oldVal = p.val;
                            if (!onlyIfAbsent)
                                p.val = value;
                        }
                    }
                }
            }
            if (binCount != 0) {
                // 如果链表中结点数量达到树化阈值(8)，尝试将链表转化为红黑树
                if (binCount >= TREEIFY_THRESHOLD)
                    treeifyBin(tab, i);
                if (oldVal != null)
                    return oldVal;
                break;
            }
        }
    }
    // 增加元素个数，若超过阈值，扩容
    addCount(1L, binCount);
    return null;
}
```

#### get(Object key)
```java
public V get(Object key) {
    Node<K,V>[] tab; Node<K,V> e, p; int n, eh; K ek;
    // 获得正数哈希值
    int h = spread(key.hashCode());
    //  哈希表不为 null，长度不为 0，对应索引的位置也不为 null 时，进入 if 语句
    if ((tab = table) != null && (n = tab.length) > 0 &&
        (e = tabAt(tab, (n - 1) & h)) != null) {
        // 键相同时返回值
        if ((eh = e.hash) == h) {
            if ((ek = e.key) == key || (ek != null && key.equals(ek)))
                return e.val;
        }
        // 哈希值小于零说明是红黑树结点或正在扩容，调用 find 方法查找
        else if (eh < 0)
            return (p = e.find(h, key)) != null ? p.val : null;
        // 在链表中查找元素
        while ((e = e.next) != null) {
            if (e.hash == h &&
                ((ek = e.key) == key || (ek != null && key.equals(ek))))
                return e.val;
        }
    }
    return null;
}
```

### 扩容
#### transfer(Node<K,V>[] tab, Node<K,V>[] nextTab)
```java
// 将元素从旧哈希表移动到新哈希表
private final void transfer(Node<K,V>[] tab, Node<K,V>[] nextTab) {
    // 这个赋值语句看着太怪了，花了好多时间，在网上找了很多资料，都没有提这个逗号的作用
    // 后来自己试着写了一下，才明白这个 stride 只是一个定义，被自己蠢哭了……
    int n = tab.length, stride;
    // stride 表示每个线程处理桶的数目
    // 如果是多核 CPU，stride 为 (哈希表大小n / (8 * CPU核数))，否则为 n
    // 但最小不会低于 16
    if ((stride = (NCPU > 1) ? (n >>> 3) / NCPU : n) < MIN_TRANSFER_STRIDE)
        stride = MIN_TRANSFER_STRIDE;
    // 如果新哈希表 nextTab 为 null，初始化新哈希表，大小为原哈希表的两倍
    if (nextTab == null) {
        try {
            @SuppressWarnings("unchecked")
            Node<K,V>[] nt = (Node<K,V>[])new Node<?,?>[n << 1];
            nextTab = nt;
        } catch (Throwable ex) {
            sizeCtl = Integer.MAX_VALUE;
            return;
        }
        nextTable = nextTab;
        // 开始转移数据的桶索引
        transferIndex = n;
    }
    // 新哈希表的大小
    int nextn = nextTab.length;
    // 扩容中的链表
    ForwardingNode<K,V> fwd = new ForwardingNode<K,V>(nextTab);
    // 标记是否需要继续尝试划分一段桶处理
    boolean advance = true;
    // 标记扩容是否结束
    boolean finishing = false;
    // bound 是当前线程负责处理的最小索引，i 是当前线程正在处理的索引
    for (int i = 0, bound = 0;;) {
        Node<K,V> f; int fh;
        while (advance) {
            int nextIndex, nextBound;
            // --i 还在 bound 内，或者 finishing 为 true，不需要再划分新的桶
            if (--i >= bound || finishing)
                advance = false;
            // 如果 transferIndex 不大于 0，表示所有的桶都已经处理完了
            else if ((nextIndex = transferIndex) <= 0) {
                i = -1;
                advance = false;
            }
            // 尝试更新 nextIndex，划分新的一段需要处理的桶
            else if (U.compareAndSwapInt
                        (this, TRANSFERINDEX, nextIndex,
                        nextBound = (nextIndex > stride ?
                                    nextIndex - stride : 0))) {
                bound = nextBound;
                i = nextIndex - 1;
                advance = false;
            }
        }
        // i < 0 表示当前线程已经处理完自己负责的桶
        if (i < 0 || i >= n || i + n >= nextn) {
            int sc;
            // 整个扩容过程结束，更新 table 和 sizeCtl
            if (finishing) {
                nextTable = null;
                table = nextTab;
                sizeCtl = (n << 1) - (n >>> 1);
                return;
            }
            // 每当有一条线程完成了自己的扩容任务，使用 CAS 尝试将 sizeCtl 减 1
            if (U.compareAndSwapInt(this, SIZECTL, sc = sizeCtl, sc - 1)) {
                // 如果当前线程不是最后一条线程，直接返回
                if ((sc - 2) != resizeStamp(n) << RESIZE_STAMP_SHIFT)
                    return;
                // 最后一条线程完成了自己的扩容任务，整个扩容过程完成
                finishing = advance = true;
                i = n;
            }
        }
        // 如果旧表当前索引处是空桶，在此处放置 ForwardingNode
        else if ((f = tabAt(tab, i)) == null)
            advance = casTabAt(tab, i, null, fwd);
        // 如果旧表当前索引处的哈希值是 MOVED，说明此处是 ForwardingNode
        else if ((fh = f.hash) == MOVED)
            advance = true;
        else {
            synchronized (f) {
                if (tabAt(tab, i) == f) {
                    Node<K,V> ln, hn;
                    // 哈希值大于 0，说明是链表结点，下面的逻辑与普通的 HashMap 基本一致
                    if (fh >= 0) {
                        int runBit = fh & n;
                        Node<K,V> lastRun = f;
                        for (Node<K,V> p = f.next; p != null; p = p.next) {
                            int b = p.hash & n;
                            if (b != runBit) {
                                runBit = b;
                                lastRun = p;
                            }
                        }
                        if (runBit == 0) {
                            ln = lastRun;
                            hn = null;
                        }
                        else {
                            hn = lastRun;
                            ln = null;
                        }
                        for (Node<K,V> p = f; p != lastRun; p = p.next) {
                            int ph = p.hash; K pk = p.key; V pv = p.val;
                            if ((ph & n) == 0)
                                ln = new Node<K,V>(ph, pk, pv, ln);
                            else
                                hn = new Node<K,V>(ph, pk, pv, hn);
                        }
                        setTabAt(nextTab, i, ln);
                        setTabAt(nextTab, i + n, hn);
                        setTabAt(tab, i, fwd);
                        advance = true;
                    }
                    // 如果是红黑树结点，采用红黑树的方法移动，逻辑也差不多，略
                    else if (f instanceof TreeBin) {
                        TreeBin<K,V> t = (TreeBin<K,V>)f;
                        TreeNode<K,V> lo = null, loTail = null;
                        TreeNode<K,V> hi = null, hiTail = null;
                        int lc = 0, hc = 0;
                        for (Node<K,V> e = t.first; e != null; e = e.next) {
                            int h = e.hash;
                            TreeNode<K,V> p = new TreeNode<K,V>
                                (h, e.key, e.val, null, null);
                            if ((h & n) == 0) {
                                if ((p.prev = loTail) == null)
                                    lo = p;
                                else
                                    loTail.next = p;
                                loTail = p;
                                ++lc;
                            }
                            else {
                                if ((p.prev = hiTail) == null)
                                    hi = p;
                                else
                                    hiTail.next = p;
                                hiTail = p;
                                ++hc;
                            }
                        }
                        ln = (lc <= UNTREEIFY_THRESHOLD) ? untreeify(lo) :
                            (hc != 0) ? new TreeBin<K,V>(lo) : t;
                        hn = (hc <= UNTREEIFY_THRESHOLD) ? untreeify(hi) :
                            (lc != 0) ? new TreeBin<K,V>(hi) : t;
                        setTabAt(nextTab, i, ln);
                        setTabAt(nextTab, i + n, hn);
                        setTabAt(tab, i, fwd);
                        advance = true;
                    }
                }
            }
        }
    }
}
```

#### helpTransfer(Node<K,V>[] tab, Node<K,V> f)
```java
final Node<K,V>[] helpTransfer(Node<K,V>[] tab, Node<K,V> f) {
    Node<K,V>[] nextTab; int sc;
    if (tab != null && (f instanceof ForwardingNode) &&
        (nextTab = ((ForwardingNode<K,V>)f).nextTable) != null) {
        int rs = resizeStamp(tab.length);
        while (nextTab == nextTable && table == tab &&
                (sc = sizeCtl) < 0) {
            if ((sc >>> RESIZE_STAMP_SHIFT) != rs || sc == rs + 1 ||
                transfer(tab, nextTab);
                break;
            }
        }
        return nextTab;
    }
    return table;
}
```

#### addCount(long x, int check)
```java
private final void addCount(long x, int check) {
    // as:累加单元数组
    // b:计数器基本值
    // s:累加后的值
    CounterCell[] as; long b, s;
    // 获得累加单元数组，如果累加单元数组不存在，直接在基本值上使用 CAS 添加，若 CAS 失败，进入 if 语句
    if ((as = counterCells) != null ||
        !U.compareAndSwapLong(this, BASECOUNT, b = baseCount, s = b + x)) {
        CounterCell a; long v; int m;
        boolean uncontended = true;
        // 如果累加单元数组不存在，或当前累加单元不存在，或累加失败，调用 fullAddCount 方法
        // 此方法与 LongAdder 中的 longAccumulate 基本一致
        if (as == null || (m = as.length - 1) < 0 ||
            (a = as[ThreadLocalRandom.getProbe() & m]) == null ||
            !(uncontended =
                U.compareAndSwapLong(a, CELLVALUE, v = a.value, v + x))) {
            fullAddCount(x, uncontended);
            return;
        }
        // 如果 check 小于等于 1，返回
        if (check <= 1)
            return;
        // 计算累加和
        s = sumCount();
    }
    if (check >= 0) {
        Node<K,V>[] tab, nt; int n, sc;
        // 如果元素个数达到扩容阈值，并且哈希表容量小于最大容量，进入循环
        // 下面的逻辑与 helpTransfer 方法基本一致
        while (s >= (long)(sc = sizeCtl) && (tab = table) != null &&
                (n = tab.length) < MAXIMUM_CAPACITY) {
            int rs = resizeStamp(n);
            if (sc < 0) {
                if ((sc >>> RESIZE_STAMP_SHIFT) != rs || sc == rs + 1 ||
                    sc == rs + MAX_RESIZERS || (nt = nextTable) == null ||
                    transferIndex <= 0)
                    break;
                if (U.compareAndSwapInt(this, SIZECTL, sc, sc + 1))
                    transfer(tab, nt);
            }
            else if (U.compareAndSwapInt(this, SIZECTL, sc,
                                            (rs << RESIZE_STAMP_SHIFT) + 2))
                transfer(tab, null);
            s = sumCount();
        }
    }
}
```
